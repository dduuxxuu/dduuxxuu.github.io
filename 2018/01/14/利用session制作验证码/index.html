利用php中的session制作验证码
验证码的运用在登录界面运用很频繁。利用session实现了验证码的验证。

  • 利用php生成一张图片
  • 随机生成验证码字符串
  • 将字符串存在session里
  • 将字符串随机添加颜色大小角度加载到图片里
  • 为验证码添加干扰元素
  • 输出图片
  • 验证端输入字符，对其与session里保存字符进行比较，输出结果

//php内容

<?php 
/*实现验证码功能*/ 
//开启session 
//启用session
session_start(); 
//创建一个大小为 100*30 的验证码图片 
$image = imagecreatetruecolor(100, 30); 
$bgcolor = imagecolorallocate($image, 255, 214, 118);
//将图片区域填充 
imagefill($image, 0, 0, $bgcolor); 
$captch_code = ''; 
for ($i = 0; $i < 4; $i++) { 
    //定义字体大小
    $fontSize = 8; 
    //随机定义字体颜色
    $fontColor = imagecolorallocate($image, rand(0, 255), rand(0, 255), rand(0, 255)); 
    //定义验证码取值，排除易混字
    $data = '3456789abcdefghijkmnpqrstuvwxy'; 
    //随机取字
    $fontContainer = substr($data, rand(0, strlen($data) - 1), 1);
    //拼接字
    $captch_code .= $fontContainer; 
    $x = ($i * 100 / 4) + rand(5, 10); 
    $y = rand(5, 10); 
    imagestring($image, $fontSize, $x, $y, $fontContainer, $fontColor); 
} 
//就生成的验证码保存到session 
$_SESSION['randcode'] = $captch_code; 
//在图片上增加点干扰元素 
for ($i = 0; $i < 300; $i++) { 
    $pointColor = imagecolorallocate($image, rand(50, 200), rand(50, 200), rand(50, 200)); imagesetpixel($image, rand(1, 99), rand(1, 29), $pointColor); 
} 
//在图片上增加线干扰元素 
for ($i = 0; $i < 8; $i++) { 
    $lineColor = imagecolorallocate($image, rand(80, 220), rand(80, 220), rand(80, 220)); imageline($image, rand(1, 99), rand(1, 29), rand(1, 99), rand(1, 29), $lineColor); 
} 
//设置头 
header('content-type:image/png'); 
imagepng($image); 
imagedestroy($image); 

?>
